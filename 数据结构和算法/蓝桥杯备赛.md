# 第三题

<img width="824" height="926" alt="image" src="https://github.com/user-attachments/assets/49fe0cc9-9c06-43f2-8c7b-9c8de6343f20" />
很恶心的一题 用到的知识点很多 应该说是i很不错的一题

首先环形字符串最大共长 正常我们用动态规划来做 但是这里是环形 因此我们选择使用两倍原字符串 然后截取长度小于l的部分 就是环形匹配了 我们使用哈希表记录所有长度为l的部分 然后匹配的时候从大到小 如果切割长度中有跟这个长度相同的 所有最大共长就这么长 

确定边权后 观察到 这题是连接n个点用n-1边 是最大生成树类型 因此用克鲁斯卡尔算法 并且用并查集判断 就能完成

```python
import os
import sys

# 增加递归深度，防止并查集 find 递归过深报错
sys.setrecursionlimit(200000)

n, m = map(int, input().split())
p = []
for _ in range(n):
    p.append(input())

# 存放所有字符串的子串字典
all_subs = []
for s in p:
    ss = s + s
    # 字典初始化
    subs = {L: set() for L in range(1, m + 1)}
    for L in range(1, m + 1):  # 统一使用大写 L
        for i in range(m):
            subs[L].add(ss[i : i+L])
    all_subs.append(subs)

edges = []
for i in range(n):
    for j in range(i + 1, n):
        weight = 0
        # 从最大可能长度 m 开始倒序尝试匹配
        for L in range(m, 0, -1):
            # 利用 Python 集合求交集的特性匹配
            if all_subs[i][L] & all_subs[j][L]:
                weight = L
                break
        # 注意这里的缩进：在 for L 循环结束后，再将权值加入边集
        edges.append((weight, i, j))

# 倒序排序，参数是 reverse=True
edges.sort(key=lambda x: x[0], reverse=True)

parent = list(range(n))

def find(i):
    if parent[i] == i:
        return i
    # 路径压缩
    parent[i] = find(parent[i])
    return parent[i]

def union(i, v):
    pi = find(i)
    pv = find(v)
    if pi == pv:
        return False
    else:
        # 【关键修复】将 i 的根节点指向 v 的根节点
        parent[pi] = pv  
    return True

ans = 0
edges_added = 0
    
# 贪心选择边
for w, u, v in edges:
    if union(u, v):
        ans += w
        edges_added += 1
        # 当加够 n-1 条边时，树已生成，可以提前退出
        if edges_added == n - 1:
            break  
            
print(ans)
```

这个函数的作用就是 在sort的时候 可以自己定下规则 如果1 就是a<b -1 就是a>b 0 就是按长度区分

因为我们只需要两两对比 然后发现哪一个顺序更大 就按这个顺序来排 这样最后排出来的就是最大值

```python
import os

# 请在此输入您的代码

  if a+b>b+a:
    return 1
  elif a+b<b+a:
    return -1
  else:
    return 0
nums.sort(key=cmp_to_key(compare))
nums=nums[::-1]

### 什么时候用二分答案呢
1. 字眼明显：题目结尾经常问**“最小化最大值”（比如：让耗时最长的那个人耗时尽可能短）、“最大化最小值”（比如：刚才那道倒水题，让水最少的瓶子里的水尽可能多），或者求某个特定的“最大/小可能值”**。
2. 答案存在一个明确的分界线。比如求“最大可行水量”：如果水量 $X$ 是可行的，那比 $X$ 小的水量肯定也都可行；
3. 正向求解极其困难：你发现从已知条件一步步推导最终答案，情况太多，完全无从下手。具备“单调性”（最核心）：如果 $X$ 不可行，那比 $X$ 大的水量肯定也都不可行。这就形成了一半是 True，一半是 False 的单调区间。

# 第九题
# 三国游戏：不等式转化与贪心算法

## 题目描述
小蓝正在玩一款游戏。游戏中魏($X$)、蜀($Y$)、吴($Z$)三个国家各自拥有一定数量的士兵 $X, Y, Z$（一开始可以认为都为 $0$）。

游戏有 $n$ 个可能会发生的事件，每个事件之间相互独立且最多只会发生一次，当第 $i$ 个事件发生时会分别让 $X, Y, Z$ 增加 $A_i, B_i, C_i$。

当游戏结束时（所有事件的发生与否已经确定），如果 $X, Y, Z$ 的其中一个大于另外两个之和，我们认为其获胜。例如，当 $X > Y + Z$ 时，我们认为魏国获胜。

小蓝想知道游戏结束时如果有其中一个国家获胜，**最多**发生了多少个事件？如果不存在任何能让某国获胜的情况，请输出 `-1`。

## 输入格式
第一行包含一个整数 $n$。
第二行包含 $n$ 个整数表示 $A_i$。
第三行包含 $n$ 个整数表示 $B_i$。
第四行包含 $n$ 个整数表示 $C_i$。

## 输出格式
输出一行包含一个整数表示答案。

## 样例输入
```text
3
1 2 2
2 3 2
1 0 7

题目理解要清楚 这一题虽然给了三个可能的情况 因此如果使用贪心或者动态规划会很不好做 但是有个很巧妙的做法 就是把每一个元素对于每种结果的权重算出来 也就是贡献度 此时只需要进行一次排序 就能很轻松的算出来了
是一个很棒的**思想**
```python
import sys

def solve():
    # 极速读取所有输入，规避数据量大时的超时风险
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    
    # 提取 A, B, C 数组
    A = [int(x) for x in input_data[1 : n+1]]
    B = [int(x) for x in input_data[n+1 : 2*n+1]]
    C = [int(x) for x in input_data[2*n+1 : 3*n+1]]
    
    # 定义一个内部函数，用于计算某个国家获胜时的最大事件数
    def get_max_events(W, L1, L2):
        # 计算该国视角的贡献值
        deltas = [W[i] - L1[i] - L2[i] for i in range(n)]
        # 贪心：降序排序
        deltas.sort(reverse=True)
        
        cur_sum = 0
        count = 0
        for d in deltas:
            cur_sum += d
            if cur_sum > 0:
                count += 1
            else:
                break
        return count

    # 分别计算三个国家获胜的最大事件数
    ans_x = get_max_events(A, B, C)
    ans_y = get_max_events(B, A, C)
    ans_z = get_max_events(C, A, B)
    
    # 取三者的最大值
    max_ans = max(ans_x, ans_y, ans_z)
    
    # 如果最大值是 0，说明没有任何一个国家能在哪怕 1 个事件发生时获胜
    if max_ans == 0:
        print("-1")
    else:
        print(max_ans)

if __name__ == '__main__':
    solve()
```

接下来 $n$ 行，每行包含 $m$ 个整数，相邻整数之间使用一个空格分隔，表示矩阵中的每个数 $A_{i,j}$。

## 输出格式

## 样例输入
```text

这题其实我是会做的 这种区间极值问题 一看就是单调队列 但是具体实现 方式有些忘了 这边回顾一下

首先 一个用来记录 一个队列用来遍历

**队列存的是下标而不是值!**

然后首先第一个while判断队列非空 以及如果索引在区间外 就popleft

然后下一个while维护队列单调 注意 最值最后是左边出来的 因此最大值的话是从大到小排

然后每个值都需要append

然后判断如果idx开始大于最小的区间了 就可以开始记录最值了

```python
import sys
from collections import deque

def solve():
    # 使用 sys.stdin.read().split() 极大地提升 Python 的输入读取速度
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    a = int(input_data[2])
    b = int(input_data[3])
    MOD = 998244353
    
    # 构建初始矩阵
    matrix = []
    idx = 4
    for i in range(n):
        row = [int(x) for x in input_data[idx : idx + m]]
        matrix.append(row)
        idx += m
        
    # 第一步：横向处理（按行处理）
    # 创建 row_max 和 row_min 矩阵，用来存储每行长度为 b 的窗口的最值
    cols_after_row_process = m - b + 1
    row_max = [[0] * cols_after_row_process for _ in range(n)]
    row_min = [[0] * cols_after_row_process for _ in range(n)]
    
    for i in range(n):
        q_max = deque()
        q_min = deque()
        for j in range(m):
            # 维护最大值单调递减队列 (存的是索引)
            while q_max and q_max[0] <= j - b:
                q_max.popleft() # 队首元素滑出窗口
            while q_max and matrix[i][q_max[-1]] <= matrix[i][j]:
                q_max.pop()     # 保持单调递减
            q_max.append(j)
            
            # 维护最小值单调递增队列 (存的是索引)
            while q_min and q_min[0] <= j - b:
                q_min.popleft() # 队首元素滑出窗口
            while q_min and matrix[i][q_min[-1]] >= matrix[i][j]:
                q_min.pop()     # 保持单调递增
            q_min.append(j)
            
            # 当窗口形成时，记录最值
            if j >= b - 1:
                row_max[i][j - b + 1] = matrix[i][q_max[0]]
                row_min[i][j - b + 1] = matrix[i][q_min[0]]
                
    # 第二步：纵向处理（按列处理）并计算最终答案
    ans = 0
    for j in range(cols_after_row_process):
        q_max = deque()
        q_min = deque()
        for i in range(n):
            # 维护列方向的最大值队列
            while q_max and q_max[0] <= i - a:
                q_max.popleft()
            while q_max and row_max[q_max[-1]][j] <= row_max[i][j]:
                q_max.pop()
            q_max.append(i)
            
            # 维护列方向的最小值队列
            while q_min and q_min[0] <= i - a:
                q_min.popleft()
            while q_min and row_min[q_min[-1]][j] >= row_min[i][j]:
                q_min.pop()
            q_min.append(i)
            
            # 当完整的 a * b 窗口形成时，计算价值并累加
            if i >= a - 1:
                cur_max = row_max[q_max[0]][j]
                cur_min = row_min[q_min[0]][j]
                # 每次乘法和加法后都要取模
                ans = (ans + cur_max * cur_min) % MOD
                
    print(ans)

if __name__ == '__main__':
    solve()
```

接下来 $T$ 行，每行包含三个整数 $n$，$m$，$k$，表示一组询问。

这是一个比较单纯的规律题 没什么算法内容

关键在于 我们不需要具体判断这个点到底在那个层 我们只需要在一一个计算 这一层的个数是右边界减去左边界+1 所以我们只需要迭代左右边界就可以 然后要求子数节点 所以迭代从这个节点开始就可以了
```python
n=int(input())
ans=[]
for _ in range(n):
  a,b,c=map(int,input().split())
  if b==1:
    ans.append(a-c+1)
    continue
  ta=0
  l=c
  r=c
  while l<=a:
    ta+=min(r,a)-l+1
    l=(l-1)*b+2
    r=r*b+1
  ans.append(ta)
for k in ans:
  print(k)
```

这一题 按我正常的思路来说 马上想到埃式筛 但是有个问题 这题的规模太大了 即便o（n）也无法满足

因此其实转变思路即可 我们不需要找到所有质数 是需要对这一个数用埃式筛算法思想 每遇到一个质数因子 就把这个数里面所有这个因子都去除 这样就防止遇到合数了 
```py
import sys
import math

def solve():
    # 读取输入
    n = int(sys.stdin.read().strip())
    ans = 0
    
    # 只需要遍历到根号n
    limit = int(math.sqrt(n))
    
    for i in range(2, limit + 1):
        # 只要能整除，i 必定是质数
        if n % i == 0:
            ans += 1
            # 核心操作：把这个质因子全部除干净！
            # 这样保证后面遇到合数时，绝对不可能满足整除条件
            while n % i == 0:
                n //= i
                
    # 如果最后 n 还大于 1，说明剩下的部分本身就是一个极大的质数
    if n > 1:
        ans += 1
        
    print(ans)

if __name__ == '__main__':
    solve()
```

这一题其实主要在于一个思路 要对一个字符串进行迭代改变 最好的做法是进行标记 然后传造一个新的字符串 然后通过标记完成新字符串的迭代 而非直接对字符串炒作

```python
s = list(input())

while True:
    new_s = []
    # 使用布尔数组标记，非常标准的做法
    flag = [True for _ in range(len(s))]
    
    # 直接用 for 循环，完美避免 i 忘记递增导致的死循环
    # 并且只遍历 1 到 len(s)-2，不用在循环里写复杂的越界判断
    for i in range(1, len(s) - 1):
        if s[i] == s[i - 1] and s[i] != s[i + 1]:
            flag[i] = flag[i + 1] = False
            # 这里不需要 continue，让它自然往下走即可
        
        if s[i] != s[i - 1] and s[i] == s[i + 1]:
            flag[i] = flag[i - 1] = False
            
    # 收集存活的字符
    for i in range(len(s)):
        if flag[i]:
            new_s.append(s[i])
            
    # 如果字符串长度不再变化，说明达到了稳态，退出循环
    if len(new_s) == len(s):
        break
    s = new_s

result = ''.join(s) if s else "EMPTY"
print(result)
```

比较值得说的一题 最明显想到的肯定是把约数求出来 可是如何分配 如何去重就是最烧脑的了

其实面对三个数 第一 我们可以固定位置 用L W去遍历约数 这样就可以把位置固定 然后面对三个数 我们完全可以使用两个数相除后的数是不是也是约数 这样下来 顺序不会乱 也能遍历到每个数 属于暴力的一种

```python
def solve():
    n = 2021041820210418
    divisors = []
    
    # 1. 找出所有约数 (时间复杂度 O(sqrt(n)))
    i = 1
    while i * i <= n:
        if n % i == 0:
            divisors.append(i)
            # 存入成对的另一个约数，防止重复当 i*i == n 时只存一个
            if i != n // i:
                divisors.append(n // i)
        i += 1
        
    ans = 0
    # 2. 在约数集合中遍历 L 和 W
    for L in divisors:
        for W in divisors:
            # 只要 L * W 能被 n 整除，H = n // (L * W) 必定也是 n 的约数
            if n % (L * W) == 0:
                ans += 1
                
    print(ans)

if __name__ == '__main__':
    solve()
```
